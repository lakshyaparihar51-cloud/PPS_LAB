#include<stdio.h>

int main()
{
   int n;
   scanf("%d",&n);
   if (n<0)
   printf("The temperature is Freezing.");
   else if(n>=0 && n<=20)
   printf("The temperature is Cold.");
   else if(n>20 && n<=35)
   printf("The temperature is Normal.");
   else
   printf("The temperature is Hot.");
   return 0;
}
